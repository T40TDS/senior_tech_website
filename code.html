          int vel[3] = {0,0,0};
          int pos[3] = {0,0,0};
          int x = A0;
          int offset;
          const int ground = A4;
          const int power = A5;
          void setup() {
            // put your setup code here, to run once:
            Serial.begin(9600);
            offset = 518;
            pinMode(ground, OUTPUT);
            pinMode(power, OUTPUT);
            digitalWrite(ground, LOW);

            digitalWrite(power, HIGH);
            Serial.print(vel[0]);
            Serial.print("\n");
            vel[0] = vel[0]-vel[0];
          }

          void loop() {
            // put your main code here, to run repeatedly:
            float signal = analogRead(x);
            float value = map(signal,offset,1024,0,100);
            //calculus time


            int test = value * 10;
            Serial.print(test);
            Serial.print("\t");
            vel[0] += value*10;
            pos[0] = pos[0] + vel[0]*10;

            int signal_I = (int)(signal);

            Serial.print(signal_I );
            Serial.print("\t");
            Serial.print(value);
            Serial.print("\t");
            Serial.print(vel[0]);
            Serial.print("\t");
            Serial.print(pos[0]);
            Serial.print("\n");


            vel[0] = 0;



            delay(10);



            }
